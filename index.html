Question 2 : "Write callback function to print numbers from 1 to 7, in which 1 should be printed after 1 sec , 2 should be printed after 2 sec, 3 should be printed after 3 sec and so on. Explain callback hell. Numbers 1 2 3 4 5 6 7"

Ans:- 

       function callBackHell(){
        setTimeout(()=>{
            console.log(add++);
            setTimeout(()=>{
                console.log(add++)
                setTimeout(()=>{
                    console.log(add++)
                    setTimeout(()=>{
                        console.log(add++)
                        setTimeout(()=>{
                            console.log(add++)
                            setTimeout(()=>{
                                console.log(add++)
                                setTimeout(()=>{
                                    console.log(add++)
                                },700)
                            },600)
                        },500)
                    },400)
                },300)
            },200)
        },1000)
       }
       let add = 1;
       callBackHell();

Question 4 : Create a promise function accepting a argument, if yes is passed to the function then it should go to resolved state and print Promise Resolved, and if nothing is passed then it should go to reject state and catch the error and print Promise Rejected

Ans:- 

       const promise = new Promise(function(res,rej){
        let a = 10;
        if(a>9){
            res("Promise Resolved");
        }
        else{
            rej("Promise Rejected");
        }
       })       
       promise.then((x)=>{
        console.log(x);
       }).catch((x)=>{
        console.log(x);
       })

Question 8 : Create examples to explain async await function.

Ans:- 

        async function f() {

            let promise = new Promise((resolve, reject) => {
              setTimeout(() => resolve("done!"), 1000)
            });
          
            let result = await promise;
          
            alert(result);
          }
          
          f();

Question 8 : Create examples to explain promise.all function.

Ans:- 


const timeOut = (t) => {
    return new Promise((resolve, reject) => {
      setTimeout(() => {
        resolve(`Completed in ${t}`)
      }, t)
    })
  }
  
  
  timeOut(1000)
   .then(result => console.log(result))
  
  Promise.all([timeOut(5000), timeOut(2000)])
   .then(result => console.log(result))
